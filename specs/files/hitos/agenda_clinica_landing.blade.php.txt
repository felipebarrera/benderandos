async function fetchPublic(url, method='GET', body=null) {
  const opts = { method, headers: {'Content-Type':'application/json','X-Requested-With':'XMLHttpRequest'} };
  if (body) opts.body = JSON.stringify(body);
  const res = await fetch(url, opts);
  const data = await res.json();
  if (!res.ok) throw new Error(data.error || data.message || `Error ${res.status}`);
  return data;
}

// ── SEMANA ────────────────────────────────────────────────────────
function getLunes(offset=0) {
  const h = new Date();
  const d = h.getDay(); // 0=dom
  const lun = new Date(h);
  lun.setDate(h.getDate() - (d === 0 ? 6 : d - 1) + offset * 7);
  lun.setHours(0,0,0,0);
  return lun;
}

function renderSemana() {
  const lunes = getLunes(estado.semanaOffset);
  const hoy   = toISO(new Date());
  const DIAS   = ['Lun','Mar','Mié','Jue','Vie','Sáb','Dom'];

  // Actualizar display de semana
  const fin = new Date(lunes); fin.setDate(lunes.getDate() + 6);
  document.getElementById('apFechaDisplay').textContent =
    lunes.toLocaleDateString('es-CL',{day:'numeric',month:'short'}) + ' – ' +
    fin.toLocaleDateString('es-CL',{day:'numeric',month:'short', year:'numeric'});

  const strip = document.getElementById('apSemana');
  strip.innerHTML = DIAS.map((d, i) => {
    const fecha = new Date(lunes);
    fecha.setDate(lunes.getDate() + i);
    const iso   = toISO(fecha);
    const esHoy = iso === hoy;
    const sel   = iso === estado.fecha;
    const pasado= fecha < new Date() && !esHoy;
    return `
    <div class="ap-dia-col ${esHoy?'hoy':''} ${sel?'sel':''} ${pasado?'pasado':''}"
         onclick="${pasado?'':''}"
         style="${pasado?'opacity:.4;pointer-events:none;':''}"
         onclick="selDia('${iso}')">
      <div class="ap-dia-lbl">${d}</div>
      <div class="ap-dia-num">${fecha.getDate()}</div>
      <div class="ap-dia-dots" id="dots-${iso}"></div>
    </div>`;
  }).join('');
}

// ── SLOTS DEL DÍA ──────────────────────────────────────────────────
async function cargarSlotsDia(fecha) {
  const body = document.getElementById('apBody');
  body.innerHTML = '<div class="ap-loading"><div class="spinner"></div> Cargando horarios disponibles…</div>';

  if (!estado.especialidad) return;

  const recursos = estado.especialidad.recursos;

  try {
    // Cargar slots para todos los médicos de la especialidad en paralelo
    const resultados = await Promise.all(recursos.map(async r => {
      // Tomar la duración del primer servicio (default 30)
      const dur = r.servicios?.[0]?.duracion_min ?? 30;
      try {
        const slots = await fetchPublic(
          `/api/public/agenda/slots?recurso_id=${r.id}&fecha=${fecha}&duracion=${dur}`
        );
        return { recurso: r, slots, error: false };
      } catch {
        return { recurso: r, slots: [], error: true };
      }
    }));

    // Dots en la semana (cantidad de slots disponibles)
    const totalSlots = resultados.reduce((s, r) => s + r.slots.length, 0);
    const dotEl = document.getElementById(`dots-${fecha}`);
    if (dotEl) {
      const ndots = Math.min(totalSlots, 4);
      dotEl.innerHTML = Array.from({length:ndots}, () => '<div class="ap-dia-dot"></div>').join('');
    }

    renderBodySlots(resultados);
  } catch(e) {
    body.innerHTML = `<div class="ap-loading" style="color:#ef4444;">Error: ${e.message}</div>`;
  }
}

function filtrarDoctor(id) {
  if (id === null) {
    estado.vistaTodos = true;
    estado.doctor = null;
    document.getElementById('btnTodos').classList.add('activo');
  } else {
    const map = getEspecialidades();
    const todos = map[estado.especialidad.nombre] || [];
    estado.doctor = todos.find(r => r.id === id) || null;
    estado.vistaTodos = false;
    document.getElementById('btnTodos').classList.remove('activo');
  }

  document.querySelectorAll('.ap-doc-item').forEach((el, i) => {
    const isSelAll = estado.vistaTodos && el.textContent.includes('Todos');
    const isSelDoc = !estado.vistaTodos && estado.doctor &&
                     el.querySelector('.ap-doc-nombre')?.textContent === estado.doctor.nombre;
    el.classList.toggle('sel', isSelAll || isSelDoc);
  });

  document.querySelectorAll('.slots-fila').forEach(f => {
    if (estado.vistaTodos) {
      f.classList.add('visible');
    } else {
      f.classList.toggle('visible', parseInt(f.dataset.recursoId) === estado.doctor?.id);
    }
  });
}

function toggleVistaTodos() {

// ── SELECCIÓN SLOT → MODAL ────────────────────────────────────────
function selSlot(recursoId, recursoNombre, horaIni, horaFin, el) {
  // Deseleccionar todo
  document.querySelectorAll('.sf-slot').forEach(s => s.classList.remove('sel'));
  el.classList.add('sel');

  // Buscar recurso completo
  const map = getEspecialidades();
  const recursos = map[estado.especialidad?.nombre] || [];
  const rec = recursos.find(r => r.id === recursoId);
  const srv = rec?.servicios?.[0] || null;

  estado.doctor     = rec;
  estado.horaInicio = horaIni;
  estado.horaFin    = horaFin;
  estado.servicio   = srv;

  // Llenar resumen
  document.getElementById('rcMedico').textContent  = recursoNombre;
  document.getElementById('rcEsp').textContent     = estado.especialidad?.nombre ?? '—';
  document.getElementById('rcFecha').textContent   = estado.fechaFmt ?? estado.fecha;
  document.getElementById('rcHora').textContent    = `${horaIni} – ${horaFin}`;
  document.getElementById('rcServicio').textContent = srv?.nombre ?? 'Consulta';
  document.getElementById('rcValor').textContent    = fmtPeso(srv?.precio ?? 0);
  document.getElementById('totalVal').textContent   = fmtPeso(srv?.precio ?? 0);
  document.getElementById('transfRef').textContent  = Math.random().toString(36).slice(2,8).toUpperCase();

  abrirModal();
}

function renderModalPaso(paso) {
  // Steps
  [1,2,3].forEach(p => {
    const el = document.getElementById(`mst${p}`);
    el.className = 'ms-step' + (p < paso ? ' done' : p === paso ? ' active' : '');
  });

  // Panels
  document.getElementById('mpaso1').style.display = paso === 1 ? 'block' : 'none';
  document.getElementById('mpaso2').style.display = paso === 2 ? 'block' : 'none';
  document.getElementById('mpaso3').style.display = paso === 3 ? 'block' : 'none';

  // Footer
  document.getElementById('btnBack').style.display = paso > 1 && paso < 3 ? 'block' : 'none';
  const btnNext = document.getElementById('btnNext');

  if (paso === 1) { btnNext.textContent = 'Continuar al pago →'; btnNext.style.display='block'; }
  if (paso === 2) { btnNext.textContent = 'Confirmar y pagar →'; btnNext.style.display='block'; }
  if (paso === 3) {
    btnNext.style.display = 'none';
    document.getElementById('btnBack').style.display = 'none';
  }
}

function modalSiguiente() {
  if (estado.modalPaso === 1) {
    const nombre = document.getElementById('fNombre').value.trim();
    const tel    = document.getElementById('fTelefono').value.trim();

    if (!tel)    { toast('Ingresa tu teléfono', 'err'); return; }
    estado.modalPaso = 2;
    renderModalPaso(2);
  } else if (estado.modalPaso === 2) {
    confirmarCita();
  }
}

function modalVolver() {

  }
}

          Tu hora quedó reservada exitosamente.<br>

            : '✅ Tu pago fue procesado correctamente.'}

          <div class="exito-card-row"><span class="ecr-lbl">Médico</span><span class="ecr-val">${estado.doctor?.nombre}</span></div>
          <div class="exito-card-row"><span class="ecr-lbl">Especialidad</span><span class="ecr-val">${estado.especialidad?.nombre}</span></div>
          <div class="exito-card-row"><span class="ecr-lbl">Fecha</span><span class="ecr-val">${estado.fechaFmt}</span></div>
          <div class="exito-card-row"><span class="ecr-lbl">Hora</span><span class="ecr-val" style="font-family:'DM Mono',monospace;">${estado.horaInicio} – ${estado.horaFin}</span></div>

          <div class="exito-card-row"><span class="ecr-lbl">Pago</span><span class="ecr-val">${pagoLabel}</span></div>

        <div class="exito-ref">Te enviaremos un recordatorio 24h antes de tu cita.</div>

    btn.textContent = 'Confirmar y pagar →';
  }
}

// ── HERO: disponibilidad hoy ──────────────────────────────────────
async function cargarHeroDisponibilidad() {
  const hoy  = toISO(new Date());
  const list = document.getElementById('hacList');

  if (!RECURSOS.length) {
    list.innerHTML = '<div style="text-align:center;color:var(--ink3);font-size:13px;padding:16px;">Sin profesionales configurados aún.</div>';
    return;
  }

  // Tomar máximo 4 recursos
  const muestra = RECURSOS.slice(0, 4);

  try {
    const resultados = await Promise.all(muestra.map(async r => {
      const dur = r.servicios?.[0]?.duracion_min ?? 30;
      try {
        const slots = await fetchPublic(`/api/public/agenda/slots?recurso_id=${r.id}&fecha=${hoy}&duracion=${dur}`);
        return { r, slots };
      } catch { return { r, slots: [] }; }
    }));

    const palette = ['#3dd9eb20','#00e5a020','#7c6af720','#ff6b3520'];
    const pText   = ['#0e7490',  '#059669',  '#6d28d9',  '#c2410c'];

    list.innerHTML = resultados.map(({r, slots}, i) => {
      const ini = r.nombre[0].toUpperCase();
      const bg  = palette[i % palette.length];
      const tc  = pText[i % pText.length];
      const proximos = slots.slice(0, 3);

      return `
      <div class="hac-row" onclick="seleccionarEspecialidad('${(r.especialidad||'General').replace(/'/g,"\\'")}')">
        <div class="hac-dot" style="background:${bg};color:${tc};">${ini}</div>
        <div class="hac-info">
          <div class="hac-name">${r.nombre}</div>
          <div class="hac-esp">${r.especialidad ?? 'General'}</div>
          ${proximos.length ? `<div class="hac-slots">${proximos.map(s=>`<span class="hac-slot">${s.hora_inicio}</span>`).join('')}</div>` : ''}
        </div>
        ${proximos.length
          ? `<div class="hac-badge-avail">${proximos.length}+ libres</div>`
          : `<div style="font-size:10px;color:var(--ink3);flex-shrink:0;">Sin horas hoy</div>`}
      </div>`;
    }).join('');
  } catch(e) {
    list.innerHTML = `<div style="color:var(--ink3);text-align:center;font-size:13px;padding:16px;">No se pudo cargar disponibilidad.</div>`;
  }
}
